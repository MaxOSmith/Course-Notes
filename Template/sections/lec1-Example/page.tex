
\begin{lstlisting}[style=Sublime-C++, caption=C++ Example]
#include <stdio.h>
#define N 10
/* Block
 * comment */
 
int main()
{
    int i;
 
    // Line comment.
    puts("Hello world!");
 
    for (i = 0; i < N; i++)
    {
        puts("LaTeX is also great for programmers!");
    }
 
    return 0;
}
\end{lstlisting}

\lstinputlisting[style=Sublime-C++, caption=Lines 6-11, firstline=6, lastline=12]{./sections/lec1-Example/ex.cpp}
