\begin{defn}[Relative Path]
	A relative path is a path that is referring to your current directory.
\end{defn}

\begin{rem}
	Recall that double quotes do not protect back quotes.
\end{rem}

#include <stdio.h>
#define N 10
/* Block
 * comment */
 
int main()
{
    int i;
 
    // Line comment.
    puts("Hello world!");
 
    for (i = 0; i < N; i++)

        puts("LaTeX is also great for programmers!");
    }
 
    return 0;
}
